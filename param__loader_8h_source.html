438  /* load helper function for generic types //{ */
439  // This function tries to load a parameter with name 'name' and a default value.
440  // You can use the flag 'optional' to not throw a ROS_ERROR when the parameter
441  // cannot be loaded and the flag 'printValues' to set whether the loaded
442  // value and name of the parameter should be printed to cout.
443  // If 'optional' is set to false and the parameter could not be loaded,
444  // the flag 'load_successful' is set to false and a ROS_ERROR message
445  // is printer.
446  // If 'unique' flag is set to false then the parameter is not checked
447  // for being loaded twice.
448  // Returns a tuple, containing either the loaded or the default value and a bool,
449  // indicating if the value was loaded (true) or the default value was used (false).
450  template <typename T>
451  std::pair<T, bool> load(const std::string& name, const T& default_value, optional_t optional = OPTIONAL, unique_t unique = UNIQUE)
452  {
453  const std::string name_prefixed = m_prefix + name;
454  T loaded = default_value;
455  if (unique && check_duplicit_loading(name_prefixed))
456  return {loaded, false};
457 
458  bool cur_load_successful = true;
459  // try to load the parameter
460  const bool success = m_nh.getParam(name_prefixed, loaded);
461  if (!success)
462  {
463  // if it was not loaded, set the default value
464  loaded = default_value;
465  if (!optional)
466  {
467  // if the parameter was compulsory, alert the user and set the flag
468  printError(std::string("Could not load non-optional parameter ") + resolved(name_prefixed));
469  cur_load_successful = false;
470  }
471  }
472 
473  if (cur_load_successful)
474  {
475  // everything is fine and just print the name_prefixed and value if required
476  if (m_print_values)
477  printValue(name_prefixed, loaded);
478  // mark the param name_prefixed as successfully loaded
479  loaded_params.insert(name_prefixed);
480  } else
481  {
482  m_load_successful = false;
483  }
484  // finally, return the resulting value
485  return {loaded, success};
486  }
