68 template <class T>
69 std::optional<T> Transformer::transformSingle(const std::string& to_frame, const T& what)
70 {
71  if (!is_initialized_)
72  {
73  ROS_ERROR_THROTTLE(1.0, "[%s]: Transformer: cannot transform, not initialized", ros::this_node::getName().c_str());
74  return std::nullopt;
75  }
76 
77  const std_msgs::Header orig_header = getHeader(what);
78  const std::string from_frame_resolved = resolveFrameName(orig_header.frame_id);
79  const std::string to_frame_resolved = resolveFrameName(to_frame);
80 
81  if (from_frame_resolved == to_frame_resolved)
82  return copyWithHeader(what, from_frame_resolved, orig_header.stamp);
83 
84  // get the transform
85  const auto tf_opt = getTransform(from_frame_resolved, to_frame_resolved, orig_header.stamp);
86  if (!tf_opt.has_value())
87  return std::nullopt;
88  mrs_lib::TransformStamped tf = tf_opt.value();
89 
90  // do the transformation
91  const auto result_opt = transform(tf, what);
92  return result_opt;
93 }

139 template <class T>
140 std::optional<T> Transformer::doTransform(const mrs_lib::TransformStamped& tf, const T& what)
141 {
142  try
143  {
144  T result;
145  geometry_msgs::TransformStamped transform = tf.getTransform();
146  tf2::doTransform(what, result, transform);
147  return result;
148  }
149  catch (tf2::TransformException& ex)
150  {
151  ROS_WARN_THROTTLE(1.0, "[%s]: Transformer: Error during transform from \"%s\" frame to \"%s\" frame.\n\tMSG: %s", node_name_.c_str(), tf.from().c_str(),
152  tf.to().c_str(), ex.what());
153  return std::nullopt;
